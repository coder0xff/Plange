tement combining a variable declaration and an assignment

Syntax 

            [
            
                type_dereference
            
            |
            
                implicit_type_dereference
            
            ]
            
                identifier
            
            "<-"
            
                expression

Example 

            <Number> x <- 100;

Example 

            <> name <- "Alice"; //implicitly type String


Example 

                object <- { "employees":[
    ("John", "Doe"),
    ("Anna", "Smith"),
    ("Peter", Jones")
]}; //type Any (duck typed)

object <- (x) { return x*2; }; //type Any
